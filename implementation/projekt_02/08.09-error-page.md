# Arbeitspaket 08.09: Error Page für App Router

## Ziel
Erstellung einer Error Page Component für das Next.js App Router System, die Fehler auf Route-Level abfängt und benutzerfreundlich darstellt.

## Kontext
- **Framework**: Next.js 15 App Router
- **Zweck**: Route-level Error Handling
- **Unterschied zu global-error.tsx**: Fängt Fehler innerhalb des Layout-Systems

## Implementierung

### Schritt 1: Root Error Page
Erstelle `apps/web/app/error.tsx`:

```typescript
'use client';

import { useEffect, useState } from 'react';
import { Button } from '@starter-kit/ui';
import { AlertTriangle, Home, RefreshCw, Bug } from 'lucide-react';
import { useRouter } from 'next/navigation';

interface ErrorPageProps {
  error: Error & { digest?: string };
  reset: () => void;
}

export default function Error({ error, reset }: ErrorPageProps) {
  const router = useRouter();
  const [errorDetails, setErrorDetails] = useState<{
    type: 'client' | 'server' | 'unknown';
    recoverable: boolean;
    suggestion: string;
  }>({ type: 'unknown', recoverable: true, suggestion: '' });

  useEffect(() => {
    // Analyze error type
    let type: 'client' | 'server' | 'unknown' = 'unknown';
    let recoverable = true;
    let suggestion = 'Bitte versuchen Sie es erneut oder kehren Sie zur Startseite zurück.';

    // Client-side errors
    if (error.message?.includes('hydration')) {
      type = 'client';
      suggestion = 'Es gab ein Problem beim Laden der Seite. Ein Neuladen sollte helfen.';
    } else if (error.message?.includes('ChunkLoadError') || error.message?.includes('Loading chunk')) {
      type = 'client';
      suggestion = 'Die Anwendung konnte nicht vollständig geladen werden. Bitte laden Sie die Seite neu.';
    } else if (error.message?.includes('Network')) {
      type = 'client';
      suggestion = 'Netzwerkfehler. Bitte überprüfen Sie Ihre Internetverbindung.';
    }
    // Server errors
    else if (error.digest) {
      type = 'server';
      suggestion = 'Ein Serverfehler ist aufgetreten. Unser Team wurde benachrichtigt.';
    }
    // Database errors
    else if (error.message?.includes('database') || error.message?.includes('prisma')) {
      type = 'server';
      recoverable = false;
      suggestion = 'Datenbankfehler. Bitte kontaktieren Sie den Support.';
    }

    setErrorDetails({ type, recoverable, suggestion });

    // Log error for monitoring
    console.error('Page error caught:', {
      message: error.message,
      digest: error.digest,
      stack: error.stack,
      type,
      url: window.location.href,
      timestamp: new Date().toISOString(),
    });

    // Send to error tracking in production
    if (process.env.NODE_ENV === 'production' && window.gtag) {
      window.gtag('event', 'exception', {
        description: error.message,
        fatal: !recoverable,
      });
    }
  }, [error]);

  return (
    <div className="min-h-screen flex items-center justify-center p-4 bg-gradient-to-b from-gray-50 to-gray-100">
      <div className="max-w-lg w-full">
        {/* Error Card */}
        <div className="bg-white rounded-2xl shadow-xl overflow-hidden">
          {/* Header with gradient */}
          <div className="bg-gradient-to-br from-red-500 to-pink-600 p-6 text-white">
            <div className="flex items-center space-x-3">
              <div className="flex-shrink-0">
                <div className="w-12 h-12 bg-white/20 rounded-full flex items-center justify-center">
                  <AlertTriangle className="w-7 h-7" />
                </div>
              </div>
              <div>
                <h1 className="text-2xl font-bold">
                  {errorDetails.type === 'server' ? 'Serverfehler' : 
                   errorDetails.type === 'client' ? 'Ladefehler' : 
                   'Ein Fehler ist aufgetreten'}
                </h1>
                <p className="text-white/90 text-sm mt-1">
                  {errorDetails.recoverable ? 
                    'Dieser Fehler kann möglicherweise behoben werden' : 
                    'Kritischer Fehler aufgetreten'}
                </p>
              </div>
            </div>
          </div>

          {/* Body */}
          <div className="p-6 space-y-4">
            {/* Error Description */}
            <div className="bg-gray-50 rounded-lg p-4">
              <h2 className="font-semibold text-gray-900 mb-2">
                Was ist passiert?
              </h2>
              <p className="text-sm text-gray-600">
                {errorDetails.suggestion}
              </p>
            </div>

            {/* Error Details (Development only) */}
            {process.env.NODE_ENV === 'development' && (
              <details className="group">
                <summary className="cursor-pointer flex items-center space-x-2 text-sm text-gray-500 hover:text-gray-700">
                  <Bug className="w-4 h-4" />
                  <span>Technische Details anzeigen</span>
                </summary>
                <div className="mt-3 bg-gray-900 rounded-lg p-4">
                  <code className="text-xs text-gray-100 block">
                    <div className="font-semibold text-red-400 mb-2">
                      {error.name || 'Error'}: {error.message}
                    </div>
                    {error.digest && (
                      <div className="text-gray-400 mb-2">
                        Digest: {error.digest}
                      </div>
                    )}
                    {error.stack && (
                      <pre className="text-gray-300 whitespace-pre-wrap overflow-auto max-h-48 mt-2">
                        {error.stack}
                      </pre>
                    )}
                  </code>
                </div>
              </details>
            )}

            {/* Actions */}
            <div className="flex flex-col sm:flex-row gap-3">
              {errorDetails.recoverable && (
                <Button
                  onClick={reset}
                  className="flex-1"
                  size="lg"
                >
                  <RefreshCw className="w-4 h-4 mr-2" />
                  Erneut versuchen
                </Button>
              )}
              <Button
                onClick={() => router.push('/')}
                variant={errorDetails.recoverable ? "outline" : "default"}
                className="flex-1"
                size="lg"
              >
                <Home className="w-4 h-4 mr-2" />
                Zur Startseite
              </Button>
            </div>

            {/* Support Information */}
            {error.digest && (
              <div className="pt-4 border-t border-gray-200">
                <p className="text-xs text-center text-gray-500">
                  Wenn das Problem weiterhin besteht, kontaktieren Sie bitte den Support mit dieser ID:
                </p>
                <div className="mt-2 flex justify-center">
                  <code className="bg-gray-100 px-3 py-1 rounded text-xs font-mono select-all">
                    {error.digest}
                  </code>
                </div>
              </div>
            )}
          </div>
        </div>

        {/* Additional Help */}
        <div className="mt-6 text-center">
          <p className="text-sm text-gray-600">
            Benötigen Sie Hilfe?{' '}
            <a
              href="/support"
              className="text-blue-600 hover:text-blue-700 font-medium"
            >
              Support kontaktieren
            </a>
          </p>
        </div>
      </div>
    </div>
  );
}
```

```typescript
'use client';

import { Button } from '@starter-kit/ui';
import { RefreshCw, ArrowLeft } from 'lucide-react';
import { useRouter } from 'next/navigation';

export default function DashboardError({
  error,
  reset,
}: {
  error: Error;
  reset: () => void;
}) {
  const router = useRouter();

  return (
    <div className="p-8">
      <div className="max-w-md mx-auto bg-red-50 border border-red-200 rounded-lg p-6">
        <h2 className="text-lg font-semibold text-red-800 mb-2">
          Dashboard konnte nicht geladen werden
        </h2>
        <p className="text-sm text-red-600 mb-4">
          {error.message || 'Ein unerwarteter Fehler ist aufgetreten'}
        </p>
        <div className="flex gap-2">
          <Button onClick={reset} size="sm">
            <RefreshCw className="w-4 h-4 mr-1" />
            Neu laden
          </Button>
          <Button 
            onClick={() => router.back()} 
            variant="outline" 
            size="sm"
          >
            <ArrowLeft className="w-4 h-4 mr-1" />
            Zurück
          </Button>
        </div>
      </div>
    </div>
  );
}
```

## Verwendung

### Beispiel 1: Trigger Error Page
```typescript
// In any page or component
throw new Error('Test error');
// Zeigt error.tsx der nächsten Route
```

### Beispiel 2: Trigger Not Found
```typescript
import { notFound } from 'next/navigation';

// In page component
if (!data) {
  notFound(); // Zeigt not-found.tsx
}
```

### Beispiel 3: Custom Error Handling
```typescript
// In page.tsx
export default async function Page() {
  try {
    const data = await fetchData();
    return <Component data={data} />;
  } catch (error) {
    // This will trigger error.tsx
    throw new Error(`Failed to load: ${error.message}`);
  }
}
```

## Potentielle Probleme

### Problem: Error Page selbst wirft Error
**Lösung**: Minimale Dependencies, try-catch in useEffect

### Problem: Hydration Errors
**Lösung**: 'use client' directive und SSR-safe code
